local function autoInteract(object)
    if not object then return end
    for _, v in ipairs(object:GetDescendants()) do
        if v:IsA("ProximityPrompt") then
            fireproximityprompt(v, 1) 
            return true
        end
    end
    return false
end

CreateToggle(MainTab, "⚔️ Auto Farm Brainrots", function(state)
    getgenv().AutoFarmBrainrot = state
    if state then
        if not getgenv().BaseReturnCFrame then
            local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
            if root then getgenv().BaseReturnCFrame = root.CFrame end
        end
        
        task.spawn(function()
            while getgenv().AutoFarmBrainrot do
                pcall(function()
                    local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                    -- แปลงชื่อเพื่อหาโฟลเดอร์ของจริง
                    local realZone = ZoneMap[getgenv().SelectedBrainrotZone]
                    local targetZone = workspace:FindFirstChild("SpawnedBrainrots") and workspace.SpawnedBrainrots:FindFirstChild(realZone)

                    if root and targetZone then
                        for _, item in pairs(targetZone:GetChildren()) do
                            if not getgenv().AutoFarmBrainrot then break end
                            local targetPos = item:IsA("BasePart") and item.CFrame or (item:IsA("Model") and item:GetPivot())
                            if targetPos then
                                root.CFrame = targetPos
                                task.wait(0.2) 
                                autoInteract(item)
                                task.wait(0.3) 
                                if getgenv().BaseReturnCFrame then
                                    root.CFrame = getgenv().BaseReturnCFrame
                                end
                                task.wait(0.3)
                            end
                        end
                    end
                end)
                task.wait(0.5)
            end
        end)
    end
end)

    local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    if root and workspace:FindFirstChild("Plots") then
        local closestPlot = nil
        local shortestDistance = math.huge
        
        for _, plot in pairs(workspace.Plots:GetChildren()) do
            local distance = (root.Position - plot:GetPivot().Position).Magnitude
            if distance < shortestDistance then
                shortestDistance = distance
                closestPlot = plot

            btn.Text = "📍 Locked: " .. closestPlot.Name
        end
    end
end)
